        }

% --- Subsequent pages style ---

    \renewcommand{\headrulewidth}{0pt} % remove top line

    \fancyfoot[L]{%
        % Citation centered
        \vspace*{1cm}
        \begin{flushleft}
        \textit{\textbf{Citation:} \VAR{citation | latex_escape}}
        \end{flushleft}

        % Bottom line: page left, journal right
        \noindent\makebox[\textwidth]{%
        \textit{\VAR{journalYearVolumeIssue}}   \hfill \thepage
        }
    }
}
